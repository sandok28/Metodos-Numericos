from numpy import *
import matplotlib.pyplot as plt
g= 9.8
m= 68.1
c= 12.5
x=[]
y1=[]

e=2.72
for i in range (60):
    v1=(g*m/c)*(1-pow(e,-(c/m)*i))
    if(i==0):
        print(0,0)
    print(i,v1)
    x.append(i)
    y1.append(v1)
plt.plot(x,y1, label="Analitico")
plt.hold(True)
plt.grid(True)
plt.legend(loc="upper left")
plt.title("Caida Libre")
plt.xlabel("Velocidad") 
plt.ylabel("Tiempo") 
plt.show()
